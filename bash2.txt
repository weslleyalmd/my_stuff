__git_ps1 ()
{
    local b="$(git symbolic-ref HEAD 2>/dev/null)";
    if [ -n "$b" ]; then
        printf " (%s)" "${b##refs/heads/}";
    fi
}

parse_git_dirty ()
{
  GSTATUS=$(git status 2> /dev/null | tail -n1)

  if [ "$GSTATUS" != "nothing to commit, working tree clean" ] && [ "$GSTATUS" != "" ]; then
  	echo "*"
  fi

}
